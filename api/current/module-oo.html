<static> extend(Sub, Super)

Sub extends Super, i.e. Sub inherits all members from Super. (Pseudoclass inheritance)

Parameters:
Name Type Description
Sub *

child
Super *

parent
Since:
  • 0.0.1
Source:
  • oo.js, line 54
See:
  • http://molily.de/weblog/javascript-pseudoklassen
Example
var oo = require('./oo.js');

function SuperClass(a) {
  this._a = a;
}

// You need to call the constructor of SuperClass,
// otherwise attributes may not get initialized
// properly.
function SubClass(a, b) {
  SuperClass.call(this, a);
  this._b = b;
}

oo.extend(SubClass, SuperClass);
